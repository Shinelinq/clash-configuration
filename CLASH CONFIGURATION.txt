function getAllProxyNames(config) {
  // åˆå¹¶æ”¶é›†ï¼ˆé‡è¦ï¼‰ï¼šé¿å…æ³¨å…¥æ‰‹åŠ¨èŠ‚ç‚¹åˆ° proxies åé®è”½ provider èŠ‚ç‚¹
  const names = [];

  // proxies å­—æ®µ
  if (Array.isArray(config.proxies) && config.proxies.length) {
    names.push(...config.proxies.map(p => p && p.name).filter(Boolean));
  }

  // proxy-providers å±•å¼€çš„ proxies
  const providers = config["proxy-providers"] || {};
  for (const key of Object.keys(providers)) {
    const p = providers[key];
    if (Array.isArray(p.proxies)) {
      names.push(...p.proxies.map(x => x && x.name).filter(Boolean));
    }
  }

  return [...new Set(names)];
}

function getProxiesByRegex(params, regex) {
  const names = params._allProxyNames || (params.proxies || []).map((e) => e.name);
  const manualNames = Array.isArray(params._manualProxyNames) ? params._manualProxyNames : [];
  const manualSet = new Set(manualNames);

  const base = (!MANUAL_JOIN_REGION_TEST && manualNames.length > 0)
    ? names.filter(n => !manualSet.has(n))
    : names;

  return base.filter((name) => regex.test(name));
}

// - é»˜è®¤æŒ‡å‘ "ğŸŒ Auto"
// - è‹¥ "ğŸŒ Auto" ä¸å­˜åœ¨ï¼Œåˆ™å›é€€åˆ° PROXY_NAME
// - ä»…ä½œç”¨äºâ€œéœ€è¦è‡ªåŠ¨è¡¥é»˜è®¤ dialerâ€çš„æ‰‹åŠ¨èŠ‚ç‚¹ï¼›ç”¨æˆ·æ˜¾å¼è®¾ç½®çš„ä¸è¦†ç›–

function finalizeManualDialerProxy(config) {
  const needNames = Array.isArray(config._manualNeedAutoDialerNames) ? config._manualNeedAutoDialerNames : [];
  if (!needNames.length) return;

  const groups = Array.isArray(config["proxy-groups"]) ? config["proxy-groups"] : [];
  const groupNames = new Set(groups.map(g => g && g.name).filter(Boolean));

  const target = groupNames.has(MANUAL_DEFAULT_DIALER) ? MANUAL_DEFAULT_DIALER : MANUAL_FALLBACK_DIALER;
  const needSet = new Set(needNames);

  if (!Array.isArray(config.proxies)) return;

  // ä»…ç»™â€œéœ€è¦è‡ªåŠ¨è¡¥é»˜è®¤ dialerâ€çš„èŠ‚ç‚¹è®¾ç½®/ä¿®æ­£ dialer-proxy
  for (const p of config.proxies) {
    if (!p || !p.name) continue;
    if (!needSet.has(p.name)) continue;
    p["dialer-proxy"] = target;
  }
}
